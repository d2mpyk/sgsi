89: def extract_document_metadata_from_filename(
90:     filename: str, default_version: str = "1.0"
91: ) -> tuple[str, str]:
92:     """
93:     Extrae un título limpio y la versión desde el nombre del archivo.
94:     Ejemplo: "Politica_Acceso_v2.3.pdf" -> ("Politica_Acceso", "2.3")
95:     Si no encuentra versión, conserva la versión por defecto.
96:     """
97:     filename_no_ext = os.path.splitext(filename)[0]
98:     version_match = re.search(r"(?:[-_\s]+)?[vV](\d+(?:\.\d+)+)", filename_no_ext)
99: 
100:     extracted_title = filename_no_ext
101:     extracted_version = default_version
102: 
103:     if version_match:
104:         extracted_version = version_match.group(1)
105:         clean_title = filename_no_ext[: version_match.start()].strip()
106:         if clean_title:
107:             extracted_title = clean_title
108: 
109:     return extracted_title, extracted_version

112: def build_document_description(
113:     original_filename: str,
114:     provided_description: Optional[str] = None,
115: ) -> str:
116:     """
117:     Construye la descripción persistida asegurando que incluya
118:     el nombre original del archivo subido.
119:     """
120:     filename_note = original_filename
121:     clean_description = (provided_description or "").strip()
122: 
123:     if not clean_description:
124:         return filename_note
125: 
126:     if filename_note in clean_description:
127:         return clean_description
128: 
129:     return f"{clean_description}\n{filename_note}"

132: def upsert_policy_download_read(
133:     db: Session,
134:     user_id: int,
135:     document_id: int,
136: ) -> DocumentRead:
137:     """
138:     Garantiza un registro en document_reads por usuario y política.
139:     - Si no existe, lo crea con download_at.
140:     - Si existe y no tiene read_at, refresca download_at.
141:     - Si ya tiene read_at, conserva download_at intacto.
142:     """
143:     existing_read = (
144:         db.execute(
145:             select(DocumentRead).where(
146:                 DocumentRead.user_id == user_id,
147:                 DocumentRead.document_id == document_id,
148:             )
149:         )
150:         .scalars()
151:         .first()
152:     )
153: 
154:     if existing_read is None:
155:         existing_read = DocumentRead(
156:             user_id=user_id,
157:             document_id=document_id,
158:             download_at=datetime.now(UTC),
159:             read_at=None,
160:         )
161:         db.add(existing_read)
162:         db.commit()
163:         db.refresh(existing_read)
164:         return existing_read
165: 
166:     if existing_read.read_at is None:
167:         existing_read.download_at = datetime.now(UTC)
168:         db.commit()
169:         db.refresh(existing_read)
170: 
171:     return existing_read

184:     """Homologa datetimes naive/aware para comparaciones consistentes."""

186:         return None

192: def svg_bar_chart(items: list[tuple[str, float]], width: int = 760, height: int = 260) -> Markup:
193:     """Genera un gráfico de barras horizontal en SVG para el reporte."""
194:     if not items:
195:         return Markup("<p>No hay datos suficientes para construir el gráfico por área.</p>")
196: 
197:     left_margin = 180
198:     top_margin = 20
199:     row_height = 32
200:     chart_width = width - left_margin - 60
201:     svg_height = max(height, top_margin + row_height * len(items) + 30)
202:     bars = []
203: 
204:     for index, (label, value) in enumerate(items):
205:         y = top_margin + index * row_height
206:         bar_width = max(0, min(chart_width, chart_width * (value / 100)))
207:         color = "#2f855a" if value >= 90 else "#d69e2e" if value >= 70 else "#c53030"
208:         bars.append(
209:             f'<text x="10" y="{y + 18}" font-size="12" fill="#334155">{label}</text>'
210:             f'<rect x="{left_margin}" y="{y + 4}" width="{chart_width}" height="16" rx="8" fill="#e2e8f0"></rect>'
211:             f'<rect x="{left_margin}" y="{y + 4}" width="{bar_width}" height="16" rx="8" fill="{color}"></rect>'
212:             f'<text x="{left_margin + chart_width + 10}" y="{y + 18}" font-size="12" fill="#0f172a">{value:.1f}%</text>'
213:         )
214: 
215:     svg = (
216:         f'<svg width="100%" viewBox="0 0 {width} {svg_height}" role="img" aria-label="Cumplimiento por área">'
217:         + "".join(bars)
218:         + "</svg>"
219:     )
220:     return Markup(svg)

249: def svg_trend_chart(points: list[PolicyAuditTrendPoint], width: int = 760, height: int = 260) -> Markup:
250:     """Genera un gráfico de tendencia en SVG."""
251:     if not points:
252:         return Markup("<p>No hay datos suficientes para construir la tendencia histórica.</p>")
253: 
254:     padding_x = 55
255:     padding_y = 25
256:     chart_width = width - (padding_x * 2)
257:     chart_height = height - (padding_y * 2)
258:     max_value = 100
259:     step_x = chart_width / max(len(points) - 1, 1)
260: 
261:     coords = []
262:     labels = []
263:     for index, point in enumerate(points):
264:         x = padding_x + (index * step_x)
265:         y = padding_y + chart_height - ((point.compliance_percentage / max_value) * chart_height)
266:         coords.append((x, y, point))
267:         labels.append(
268:             f'<text x="{x}" y="{height - 8}" text-anchor="middle" font-size="11" fill="#64748b">{point.period_label}</text>'
269:         )
270: 
271:     polyline = " ".join(f"{x},{y}" for x, y, _ in coords)
272:     circles = "".join(
273:         f'<circle cx="{x}" cy="{y}" r="4" fill="#1d4ed8"></circle>'
274:         f'<text x="{x}" y="{y - 10}" text-anchor="middle" font-size="11" fill="#0f172a">{point.compliance_percentage:.1f}%</text>'
275:         for x, y, point in coords
276:     )
277:     grid = "".join(
278:         f'<line x1="{padding_x}" y1="{padding_y + (chart_height * idx / 4)}" '
279:         f'x2="{width - padding_x}" y2="{padding_y + (chart_height * idx / 4)}" stroke="#e2e8f0" stroke-width="1"></line>'
280:         for idx in range(5)
281:     )
282: 
283:     svg = (
284:         f'<svg width="100%" viewBox="0 0 {width} {height}" role="img" aria-label="Tendencia de cumplimiento">'
285:         f'{grid}'
286:         f'<polyline fill="none" stroke="#1d4ed8" stroke-width="3" points="{polyline}"></polyline>'
287:         f'{circles}'
288:         f'{"".join(labels)}'
289:         "</svg>"
290:     )
291:     return Markup(svg)

294: def build_policy_reading_audit_report(
295:     db: Session,
296:     responsible_username: str,
297: ) -> dict:
298:     """Construye el informe global de confirmación de lectura listo para auditoría."""
299:     generated_at = datetime.now(UTC)
300:     due_limit = generated_at - timedelta(days=READ_DUE_DAYS)
301: 
302:     active_users = (
303:         db.execute(
304:             select(User)
305:             .where(User.is_active == True)
306:             .order_by(User.id)
307:         )
308:         .scalars()
309:         .all()
310:     )
311:     active_policies = (
312:         db.execute(
313:             select(Document)
314:             .where(Document.doc_type == "policy", Document.is_active == True)
315:             .order_by(*DOCUMENT_SORT_ORDER)
316:         )
317:         .scalars()
318:         .all()
319:     )
320:     reads = (
321:         db.execute(
322:             select(DocumentRead)
323:             .join(Document, DocumentRead.document_id == Document.id)
324:             .where(Document.doc_type == "policy", Document.is_active == True)
325:         )
326:         .scalars()
327:         .all()
328:     )
329:     reads_map = {(read.user_id, read.document_id): read for read in reads}
330: 
331:     total_active_users = len(active_users)
332:     total_active_policies = len(active_policies)
333:     total_assignments = total_active_users * total_active_policies
334:     completed_assignments = 0
335:     pending_reads = 0
336:     overdue_reads = 0
337:     confirmed_users = set()
338:     policy_rows: list[PolicyAuditByPolicy] = []
339:     department_totals: dict[int, dict] = {}
340:     traceability_rows: list[PolicyAuditTraceabilityRow] = []
341: 
342:     for user in active_users:
343:         if user.department_id not in department_totals:
344:             department_totals[user.department_id] = {
345:                 "department_id": user.department_id,
346:                 "department_name": user.department_name,
347:                 "total_collaborators": 0,
348:                 "total_assignments": 0,
349:                 "confirmations": 0,
350:                 "pending_reads": 0,
351:                 "overdue_reads": 0,
352:             }
353:         department_totals[user.department_id]["total_collaborators"] += 1
354: 
355:     for policy in active_policies:
356:         policy_confirmations = 0
357:         policy_overdue = 0
358: 
359:         for user in active_users:
360:             read = reads_map.get((user.id, policy.id))
361:             normalized_policy_created_at = normalize_datetime(policy.created_at)
362:             normalized_read_at = normalize_datetime(read.read_at) if read else None
363:             normalized_download_at = normalize_datetime(read.download_at) if read else None
364:             is_confirmed = bool(read and normalized_read_at is not None)
365:             is_overdue = bool(normalized_policy_created_at and normalized_policy_created_at <= due_limit and not is_confirmed)
366: 
367:             department_bucket = department_totals[user.department_id]
368:             department_bucket["total_assignments"] += 1
369: 
370:             if is_confirmed:
371:                 completed_assignments += 1
372:                 policy_confirmations += 1
373:                 department_bucket["confirmations"] += 1
374:                 confirmed_users.add(user.id)
375:                 status_value = "leido"
376:             else:
377:                 pending_reads += 1
378:                 department_bucket["pending_reads"] += 1
379:                 status_value = "pendiente"
380:                 if is_overdue:
381:                     overdue_reads += 1
382:                     policy_overdue += 1
383:                     department_bucket["overdue_reads"] += 1
384:                     status_value = "fuera_de_plazo"
385: 
386:             traceability_rows.append(
387:                 PolicyAuditTraceabilityRow(
388:                     user_id=user.id,
389:                     username=user.username,
390:                     department_name=user.department_name,
391:                     policy_code=policy.code or f"POL-{policy.id}",
392:                     policy_title=policy.title,
393:                     status=status_value,
394:                     created_at=normalized_policy_created_at or policy.created_at,
395:                     download_at=normalized_download_at,
396:                     read_at=normalized_read_at,
397:                     overdue=is_overdue,
398:                 )
399:             )
400: 
401:         policy_percentage = round(
402:             (policy_confirmations / total_active_users) * 100, 1
403:         ) if total_active_users else 0.0
404:         policy_rows.append(
405:             PolicyAuditByPolicy(
406:                 policy_id=policy.id,
407:                 code=policy.code or f"POL-{policy.id}",
408:                 title=policy.title,
409:                 total_collaborators=total_active_users,
410:                 confirmations=policy_confirmations,
411:                 compliance_percentage=policy_percentage,
412:                 overdue_reads=policy_overdue,
413:                 semaphore=get_compliance_semaphore(policy_percentage),
414:             )
415:         )
416: 
417:     department_rows = []
418:     for department_data in sorted(department_totals.values(), key=lambda item: item["department_name"]):
419:         compliance = round(
420:             (department_data["confirmations"] / department_data["total_assignments"]) * 100, 1
421:         ) if department_data["total_assignments"] else 0.0
422:         department_rows.append(
423:             PolicyAuditByDepartment(
424:                 department_id=department_data["department_id"],
425:                 department_name=department_data["department_name"],
426:                 total_collaborators=department_data["total_collaborators"],
427:                 total_assignments=department_data["total_assignments"],
428:                 confirmations=department_data["confirmations"],
429:                 compliance_percentage=compliance,
430:                 pending_reads=department_data["pending_reads"],
431:                 overdue_reads=department_data["overdue_reads"],
432:                 semaphore=get_compliance_semaphore(compliance),
433:             )
434:         )
435: 
436:     global_compliance_percentage = round(
437:         (completed_assignments / total_assignments) * 100, 1
438:     ) if total_assignments else 0.0
439:     collaborators_confirmed_percentage = round(
440:         (len(confirmed_users) / total_active_users) * 100, 1
441:     ) if total_active_users else 0.0
442: 
443:     summary = PolicyAuditSummary(
444:         generated_at=generated_at,
445:         report_version="1.0",
446:         responsible=responsible_username,
447:         global_compliance_percentage=global_compliance_percentage,
448:         collaborators_confirmed_percentage=collaborators_confirmed_percentage,
449:         pending_reads=pending_reads,
450:         overdue_reads=overdue_reads,
451:         total_active_users=total_active_users,
452:         total_active_policies=total_active_policies,
453:         total_assignments=total_assignments,
454:         completed_assignments=completed_assignments,
455:     )
456: 
457:     monthly_points = []
458:     for offset in range(5, -1, -1):
459:         reference = generated_at - timedelta(days=30 * offset)
460:         month_start = reference.replace(day=1, hour=0, minute=0, second=0, microsecond=0)
461:         if month_start.month == 12:
462:             next_month = month_start.replace(year=month_start.year + 1, month=1)
463:         else:
464:             next_month = month_start.replace(month=month_start.month + 1)
465:         month_end = next_month - timedelta(microseconds=1)
466: 
467:         policies_available = [
468:             policy
469:             for policy in active_policies
470:             if normalize_datetime(policy.created_at) and normalize_datetime(policy.created_at) <= month_end
471:         ]
472:         assignments_available = total_active_users * len(policies_available)
473:         confirmed_assignments = 0
474: 
475:         if assignments_available:
476:             for policy in policies_available:
477:                 for user in active_users:
478:                     read = reads_map.get((user.id, policy.id))
479:                     normalized_read_at = normalize_datetime(read.read_at) if read else None
480:                     if normalized_read_at and normalized_read_at <= month_end:
481:                         confirmed_assignments += 1
482: 
483:         monthly_percentage = round(
484:             (confirmed_assignments / assignments_available) * 100, 1
485:         ) if assignments_available else 0.0
486:         monthly_points.append(
487:             PolicyAuditTrendPoint(
488:                 period_label=month_start.strftime("%b %Y"),
489:                 compliance_percentage=monthly_percentage,
490:                 confirmed_assignments=confirmed_assignments,
491:                 total_assignments=assignments_available,
492:             )
493:         )
494: 
495:     global_pie_chart = svg_pie_chart(
496:         global_compliance_percentage,
497:         max(0.0, round(100 - global_compliance_percentage, 1)),
498:     )
499:     department_bar_chart = svg_bar_chart(
500:         [(row.department_name, row.compliance_percentage) for row in department_rows]
501:     )
502:     trend_chart = svg_trend_chart(monthly_points)
503: 
504:     risks = []
505:     if overdue_reads > 0:
506:         risks.append(
507:             "Existen lecturas fuera de plazo que debilitan la evidencia de concienciación oportuna del SGSI."
508:         )
509:     if global_compliance_percentage < 90:
510:         risks.append(
511:             "El nivel de cumplimiento global es inferior al objetivo recomendado de control operacional."
512:         )
513:     if not risks:
514:         risks.append(
515:             "No se identifican incumplimientos críticos en la fecha de emisión; se recomienda mantener el seguimiento periódico."
516:         )
517: 
518:     recommendations = [
519:         "Reforzar recordatorios automáticos para políticas con más de 15 días sin confirmación.",
520:         "Priorizar seguimiento a departamentos con semáforo amarillo o rojo hasta recuperar el nivel objetivo.",
521:         "Incluir revisión mensual del reporte en comité SGSI como evidencia de monitoreo continuo.",
522:     ]
523: 
524:     return {
525:         "summary": summary,
526:         "policy_rows": policy_rows,
527:         "department_rows": department_rows,
528:         "traceability_rows": traceability_rows,
529:         "trend_points": monthly_points,
530:         "global_pie_chart": global_pie_chart,
531:         "department_bar_chart": department_bar_chart,
532:         "trend_chart": trend_chart,
533:         "scope_text": (
534:             "El informe cubre todas las políticas activas del SGSI vigentes en el sistema y la población de colaboradores activos con acceso al portal."
535:         ),
536:         "methodology_text": (
537:             f"Se considera lectura confirmada cuando existe un registro con fecha `read_at`; pendiente cuando no existe confirmación; y fuera de plazo cuando la política supera {READ_DUE_DAYS} días desde su publicación sin confirmación."
538:         ),
539:         "evidence_text": (
540:             "La confirmación se soporta en registros del sistema SGSI almacenados en `document_reads`, vinculados con usuario, política, fecha de descarga y fecha de aceptación."
541:         ),
542:         "risks": risks,

1090: def calculate_next_report_version(db: Session, report_code: str) -> str:
1091:     """Calcula la siguiente versión secuencial del reporte (1.0, 2.0, ...)."""
1092:     previous_docs = (
1093:         db.execute(
1094:             select(Document)
1095:             .where(Document.code == report_code)
1096:             .order_by(Document.created_at.desc(), Document.id.desc())
1097:         )
1098:         .scalars()
1099:         .all()
1100:     )
1101:     if not previous_docs:
1102:         return "1.0"
1103: 
1104:     max_version = 0
1105:     for existing in previous_docs:
1106:         match = re.match(r"^(\d+)(?:\.(\d+))?$", existing.version or "")
1107:         if not match:
1108:             continue
1109:         major = int(match.group(1))
1110:         max_version = max(max_version, major)
1111: 
1112:     if max_version == 0:
1113:         return f"{len(previous_docs) + 1}.0"
1114:     return f"{max_version + 1}.0"
